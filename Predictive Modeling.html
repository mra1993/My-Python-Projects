from sklearn.preprocessing import KBinsDiscretizer
from sklearn.linear_model import LogisticRegression

# Bin the continuous target values into discrete classes
kbins = KBinsDiscretizer(n_bins=3, encode='ordinal', strategy='uniform')
y_train_binned = kbins.fit_transform(y_train.values.reshape(-1, 1)).ravel()
y_test_binned = kbins.transform(y_test.values.reshape(-1, 1)).ravel()

scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test = scaler.transform(X_test)
selector = SelectKBest(score_func=f_classif, k=3)
X_train_selected = selector.fit_transform(X_train, y_train_binned)
X_test_selected = selector.transform(X_test)
model = LogisticRegression()
model.fit(X_train_selected, y_train_binned)
y_pred = model.predict(X_test_selected)
accuracy = accuracy_score(y_test_binned, y_pred)
conf_matrix = confusion_matrix(y_test_binned, y_pred)
class_report = classification_report(y_test_binned, y_pred)
print(f"Accuracy: {accuracy}")
print("Confusion Matrix:")
print(conf_matrix)
print("Classification Report:")
print(class_report)
